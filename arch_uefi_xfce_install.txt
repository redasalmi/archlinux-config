// Simple Steps to install Arch Linux in UEFI mode
// with the Xfce Desktop Environment

Set the keyboard layout:
	loadkeys fr-latin1


Update the system clock:
	timedatectl set-ntp true
	timedatectl status


Partition the disks:
	/dev/sda1 => 550M => EFI partition	(boot)
	/dev/sda2 => 4G => linux swap		(swap)
	/dev/sda3 => 60G => linux partition	(root)


Format the partitions:
	mkfs.fat -F32 /dev/sda1
	mkswap /dev/sda2
	swapon /dev/sda2
	mkfs.ext4 /dev/sda3


Mount the file system:
	mount /dev/sda3 /mnt


Installation:
	pacstrap /mnt base base-devel linux linux-firmware
	genfstab -U /mnt >> /mnt/etc/fstab
	arch-chroot /mnt
	ln -sf /usr/share/zoneinfo/Africa/Algiers /etc/localtime
	hwclock --systohc
	pacman -S nano
	nano /etc/locale.gen
		Uncomment en_US.UTF-8 UTF-8
	locale-gen
	nano /etc/vconsole.conf
		KEYMAP=fr-latin1


Network configuration:
	nano /etc/hostname
		myhostname
	nano /etc/hosts
		127.0.0.1	localhost
		::1			localhost
		127.0.1.1	myhostname.localdomain	myhostname


Root password:
	passwd


Boot loader:
	pacman -S grub efibootmgr
	mkdir /boot/EFI
	mount /dev/sda1 /boot/EFI
	grub-install --target=x86_64-efi --efi-directory=/boot/EFI --bootloader-id=GRUB
	grub-mkconfig -o /boot/grub/grub.cfg
	

CPU microcode:	
	intel cpu:
		pacman -S intel-ucode
	amd cpu:
		pacman -S amd-ucode


Users and groups:
	useradd -m -g users -G wheel -s /bin/bash username
	passwd username
	EDITOR=nano visudo
		Uncomment this line: # %wheel ALL=(ALL) ALL


NetworkManager:
	pacman -S networkmanager
		systemctl enable NetworkManager.service


Reboot:
	exit 
	umount -R /mnt
	reboot


Graphical user interface:
	pacman -S xorg-server xfce4 xfce4-goodies lightdm lightdm-gtk-greeter lightdm-gtk-greeter-settings
	NumLock on by default:
		pacman -S numlockx
		nano /etc/lightdm/lightdm.conf
			[Seat:*]
			greeter-setup-script=/usr/bin/numlockx on
	systemctl enable lightdm.service
	reboot


Pacman discard unused packages weekly:
	pacman -S pacman-contrib
	systemctl start paccache.timer
	systemctl enable paccache.timer


Fonts:
	pacman -S ttf-dejavu ttf-roboto ttf-liberation noto-fonts adobe-source-han-sans-otc-fonts noto-fonts-emoji


File manager functionality:
	pacman -S gvfs gvfs-mtp ntfs-3g exfat-utils
	

Audio driver:
	pacman -S pulseaudio pulseaudio-alsa pavucontrol


Video driver:
	nvidia gpu:
		pacman -S nvidia nvidia-utils nvidia-settings
	
	
Monitoring:
	pacman -S htop nethogs
	
	
Archiver:
	pacman -S xarchiver 
	pacman -S arj binutils bzip2 cpio gzip lhasa lrzip lz4   
	pacman -S lzip lzop p7zip tar unarj unrar unzip xz zip zstd
	
	
Useful Apps:
	pacman -S firefox libreoffice-fresh libreoffice-fresh-en-gb libreoffice-fresh-fr discord terminator
	pacman -S smplayer gimp poppler-glib ghostscript gparted git evince alacarte sqlitebrowser network-manager-applet
	pacman -S flameshot neofetch viewnior qbittorrent picard galculator catfish youtube-dl 


Python:
	pacman -S python tk tcl


Printer:
	pacman -S cups cups-pdf gutenprint
	systemctl enable org.cups.cupsd.service
	systemctl start org.cups.cupsd.service


Install from aur:
	git clone https://aur.archlinux.org/yay.git
		cd yay
		makepkg -si
	yay -S google-chrome visual-studio-code-bin xdman nvm postman-bin oh-my-bash-git ocs-url
	

Theme & icons: 
	install theme with ocs-url from https://www.xfce-look.org/browse/cat/
	theme: orchis-dark
	icons: tela-black
	cursor: pacman -S capitaine-cursors


Grub Customization:
	pacman -S grub-theme-vimix
	nano /etc/default/grub	
		GRUB_GFXMODE=1920x1080x32,1024x768x32,auto
		GRUB_THEME="/boot/grub/themes/Vimix/theme.txt"
		grub-mkconfig -o /boot/grub/grub.cfg


Terminal Customization:
	pacman -S bash-completion
	nano /usr/share/oh-my-bash/bashrc ~/.bashrc
	nano ~/.bashrc
		OSH_THEME="agnoster"
	nvm config:
		nano ~/.bashrc, and add this line:
			source /usr/share/nvm/init-nvm.sh
	bash
